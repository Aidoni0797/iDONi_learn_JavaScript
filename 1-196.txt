Предварительна проверка

Довольно разговоров про объекты! Давайте создадим объект и проверим его, передав функции prequal.
Откройте простейшую страницу HTML (prequeal.html),  включите в нее приведенный ниже код, загрузите
страницу и посмотрите, пройдет ли проверку объект taxi:

var taxi = {
	make: "Webville Motors",
	model: "Taxi",
	year: 1955,
	color: "yellow",
	passengers: 4,
	convertible: false,
	mileage: 281341
};

function prequal(car){
	if (car.milage>10000){
		return false;
	} else if (car.year > 1960) {
		return false;
	}
	return true;
}
	
var worthALook = prequal(taxi);

if (worthALook){
	console.log("You gotta check out this "+taxi.make+" "+taxi.model);
} else {
	console.log("You should really pass on the "+taxi.make+" "+taxi.model);
}
